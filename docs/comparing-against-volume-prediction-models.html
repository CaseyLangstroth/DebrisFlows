4.2 Get Standard Deviations

recurrence interval

risd = gartnervol[,-c(3)]
risd$diff = NA
rows = c(1:31)
for (i in rows){
  risd[i,4] = (risd[i,2]-risd[i,3])
}
g14_out = boxplot.stats(risd$diff, coef = 1)$out
g14_outind = which(risd$diff %in% c(g14_out))
risddev = sd(na.omit(risd$diff))
